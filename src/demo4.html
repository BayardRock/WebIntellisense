This demo uses CodeMirror intellisense. Declaration trigger on the key `.`. Different documentation is shown for each declaration. The data shown is hard coded.

Type `.` or `(`
// codemirror var editor4 = CodeMirror(document.getElementById('editor4'), { value: "let x = 1", mode: 'text/x-fsharp', lineNumbers: true }); var intellisense4 = new CodeMirrorIntellisense(editor4); intellisense4.addDeclarationTrigger({ keyCode: 190 }); intellisense4.addMethodsTrigger({ keyCode: 57, shiftKey: true }); intellisense4.addMethodsTrigger({ keyCode: 48, shiftKey: true }); intellisense4.addMethodsTrigger({ keyCode: 8 }); intellisense4.onMethod(function (item) { if (item.keyCode === 48 || item.keyCode === 8) { intellisense4.getMeths().setVisible(false); } else { intellisense4.setMethods(['CompareTo(int)', 'CompareTo(Object)']); } }); intellisense4.onDeclaration(function (item) { var data = [ { glyph: 3, name: 'CompareTo', documentation: 'Converts to object to another object' }, { glyph: 3, name: 'ToString', documentation: 'Converts to object to a string' } ]; intellisense4.setDeclarations(data); });
